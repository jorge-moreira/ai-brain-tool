# AI Brain Desktop App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Bundle the AI Brain CLI into a cross-platform Electron desktop app with shared UI components, while maintaining the existing npm package distribution.

**Architecture:** Monorepo using Bun workspaces with three packages: `cli` (npm package), `electron` (desktop app), and `ui` (shared React components). Core business logic stays in `packages/cli/src/core/` and is imported by Electron main process.

**Tech Stack:** Electron, React, TypeScript, Vite, shadcn/ui, electron-builder, electron-updater, Bun workspaces

        outline: "border border-input bg-background hover:bg-accent hover:text-accent-foreground",
        secondary: "bg-secondary text-secondary-foreground hover:bg-secondary/80",

        sm: "h-9 rounded-md px-3",
        lg: "h-11 rounded-md px-8",
        icon: "h-10 w-10",
      },
    },
    defaultVariants: {
      variant: "default",
      size: "default",
    },
  }
)

export interface ButtonProps
  extends React.ButtonHTMLAttributes<HTMLButtonElement>,
    VariantProps<typeof buttonVariants> {
  asChild?: boolean
}

const Button = React.forwardRef<HTMLButtonElement, ButtonProps>(
  ({ className, variant, size, asChild = false, ...props }, ref) => {
    const Comp = asChild ? Slot : "button"
    return (
      <Comp
        className={cn(buttonVariants({ variant, size, className }))}
        ref={ref}
        {...props}
      />
    )
  }
)
Button.displayName = "Button"

export { Button, buttonVariants }

// packages/cli/src/core/config.ts
import { homedir } from 'os'
import { join } from 'path'
import { existsSync, readFileSync, writeFileSync } from 'fs'

export interface BrainConfig {
  brainId: string
  obsidianPath: string
  gitRemote: string
}

const CONFIG_DIR = join(homedir(), '.ai-brain')
const CONFIG_FILE = join(CONFIG_DIR, 'config.json')

export function getConfig(): BrainConfig | null {
  if (!existsSync(CONFIG_FILE)) {
    return null
  }
  
  const content = readFileSync(CONFIG_FILE, 'utf-8')
  return JSON.parse(content)
}

export function saveConfig(config: BrainConfig): void {
  if (!existsSync(CONFIG_DIR)) {
    // Create directory
  }
  writeFileSync(CONFIG_FILE, JSON.stringify(config, null, 2))
}

export function getBrainId(): string | null {
  const config = getConfig()
  return config?.brainId || process.env.BRAIN_ID || null
}

// packages/electron/src/renderer/src/App.tsx
import { useEffect, useState } from 'react'
import { Button } from '@ai-brain/ui'
import { Card, CardHeader, CardTitle, CardContent, CardDescription } from '@ai-brain/ui'

interface BrainStatus {
  version: string
  graphExists: boolean
  gitSynced: boolean
  mcpConnected: boolean
}

export default function App() {
  const [status, setStatus] = useState<BrainStatus | null>(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    loadStatus()
  }, [])

  async function loadStatus() {
    try {
      const result = await window.electronAPI.getStatus()
      if (result.success) {
        setStatus(result.data)
      }
    } catch (error) {
      console.error('Failed to load status:', error)
    } finally {
      setLoading(false)
    }
  }

  if (loading) {
    return <div className="p-4">Loading...</div>
  }

  return (
    <div className="p-8">
      <h1 className="text-2xl font-bold mb-6">AI Brain Dashboard</h1>
      
      <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-4">
        <Card>
          <CardHeader>
            <CardTitle>Version</CardTitle>
            <CardDescription>Current version</CardDescription>
          </CardHeader>
          <CardContent>
            <p className="text-2xl font-bold">{status?.version}</p>
          </CardContent>
        </Card>

        <Card>
          <CardHeader>
            <CardTitle>Graph</CardTitle>
            <CardDescription>Knowledge graph status</CardDescription>
          </CardHeader>
          <CardContent>
            <p className={`text-2xl font-bold ${status?.graphExists ? 'text-green-600' : 'text-red-600'}`}>
              {status?.graphExists ? '✓ Exists' : '✗ Missing'}
            </p>
          </CardContent>
        </Card>

        <Card>
          <CardHeader>
            <CardTitle>Git Sync</CardTitle>
            <CardDescription>Git synchronization</CardDescription>
          </CardHeader>
          <CardContent>
            <p className={`text-2xl font-bold ${status?.gitSynced ? 'text-green-600' : 'text-yellow-600'}`}>
              {status?.gitSynced ? '✓ Synced' : '⚠ Unsynced'}
            </p>
          </CardContent>
        </Card>

        <Card>
          <CardHeader>
            <CardTitle>MCP</CardTitle>
            <CardDescription>MCP connection</CardDescription>
          </CardHeader>
          <CardContent>
            <p className={`text-2xl font-bold ${status?.mcpConnected ? 'text-green-600' : 'text-red-600'}`}>
              {status?.mcpConnected ? '✓ Connected' : '✗ Disconnected'}
            </p>
          </CardContent>
